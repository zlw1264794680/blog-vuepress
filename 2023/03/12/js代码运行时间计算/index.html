zlw
2023-03-12

# 浏览器内置方法

console.time、console.timeEnd

function fibonacci(n) {
    if (n == 1 || n == 2) {
        return 1
    };
    return fibonacci(n - 2) + fibonacci(n - 1);
}
console.time('普通递归')
fibonacci(30)
console.timeEnd('普通递归')
// 普通递归: 6.00390625 ms

# 日期对象 Date

日期对象，实现代码运行时间计算有多种写法

# Date.now

function fibonacci(n) {
    if (n == 1 || n == 2) {
        return 1
    };
    return fibonacci(n - 2) + fibonacci(n - 1);
}
const t1 = Date.now()
fibonacci(30)
const t2 = Date.now()
console.log('普通递归：', t2 - t1, t1, t2)

# new Date().getTime

function fibonacci(n) {
    if (n == 1 || n == 2) {
        return 1
    };
    return fibonacci(n - 2) + fibonacci(n - 1);
}
const t1 = new Date().getTime()
fibonacci(30)
const t2 = new Date().getTime()
console.log('普通递归：', t2 - t1, t1, t2)
