手写 Promise，符合Promise A+规范的核心内容即可。

符合 PromiseA+ 规范的 likePromise 对象与 Promise 对象是可以互操作的，这点很重要！！！包括async、await语法糖

# Promise A+规范

Promise A+规范：是对象，且有then函数。

# 使用场景

脚手架，对于低版本兼容 Promise，编译时，会判断当前环境是否拥有 Promise，没有就使用自己手写的。

# tip

globalThis 在浏览器环境指向 window对象，在 node 环境，指向 global对象

# 基础版

符合Promise A+规范即可

/**
 * 使用es6的class类，类里面的代码，会处于严格模式的状态下，直接调用方法，this指向undefined；非严格模式下，则指向window；
 */

/**
 * 符合Promise A+规范即可
 * Promise A+规范：是对象，且有then函数
 */
class MyPromise {

    /**
     * state的状态集合，不可修改
     */
    static _stateKeys = {
        PENDING: 'pending',
        FULFILLED: 'fulfilled',
        REJECTED: 'rejected'
    }

    /**
     * 任务状态
     * 规范：以下划线开头的变量，默认为私有变量，不建议外部人员使用
     */
    _state = MyPromise._stateKeys.PENDING

    /**
     * 任务状态的值
     */
    _value = void 0

    /**
     * 内部执行栈
     */
    _handlerStack = []


    /**
     * 构造函数，立即执行
     * @param {Function} executor 执行器
     */
    constructor(executor) {
        // 冻结state集合，不可修改
        Object.freeze(MyPromise._stateKeys)

        // console.log('开始状态', this._state);

        // 立即执行，处理执行器执行期间报错，走向失败状态
        try {
            executor(this._resolve.bind(this), this._reject.bind(this))
        } catch (error) {
            this._reject(error)
        }
    }

    /**
     * 改变任务状态
     * @param {string} newState 新状态
     * @param {any} value 任务传递的值
     * @returns 
     */
    _changeState(newState, value) {
        if (this._state !== MyPromise._stateKeys.PENDING) return
        this._state = newState
        this._value = value
        this._clearHandlerStackPushMicroStack()
    }

    /**
     * 任务走向成功，需要多处理参数的值类型
     * @param {any} data
     */
    _resolve(data) {
        if (MyPromise.isPromise(data)) {
            data.then((val) => {
                this._resolve(val)
            }, (reason) => {
                this._reject(reason)
            })
        } else {
            this._changeState(MyPromise._stateKeys.FULFILLED, data)
        }
    }

    /**
     * 任务走向失败，不需要多处理
     * @param {any} data
     */
    _reject(reason) {
        this._changeState(MyPromise._stateKeys.REJECTED, reason)
    }

    /**
     * 添加任务对象到执行栈
     * @param { {
     *  state:string,
     *  executor:Function,
     *  resolve:Function,
     *  reject:Function
     * } } pro 任务对象
     */
    _addProToStack(pro) {
        this._handlerStack.push(pro)
    }

    /**
     * 添加任务执行函数到微队列去
     * 参考vue源码的写法
     * @param {Function} callback
     */
    _addMicroStack(callback) {

        if (globalThis.process && globalThis.process.nextTick) {
            // 判断node环境
            process.nextTick(callback)
        } else if (MutationObserver) {
            // 浏览器环境有没有MutationObserver函数
            const p = document.createElement('p')
            const observer = new MutationObserver(callback)
            observer.observe(p, {
                childList: true // 观察该元素内部的变化
            })
            p.textContent = '1'
        } else {
            // 下下之策，推入的是宏队列
            setTimeout(callback, 0)
        }
    }

    /**
     * 清空内部执行栈，将执行器推入微队列
     * 
     * 每次任务已决，相对的所有下一级任务执行就都可以加入微队列
     */
    _clearHandlerStackPushMicroStack() {
        if (this._state === MyPromise._stateKeys.PENDING) return
        for (const p of this._handlerStack) {
            this._pushMicroStack(p)
        }
        this._handlerStack = []
    }

    /**
     * 推入微队列
     * @param {{
     *  state:string,
     *  executor:Function,
     *  resolve:Function,
     *  reject:Function
     * }} handler
     */
    _pushMicroStack(handler) {
        const { state, executor, resolve, reject } = handler
        // 过滤状态不一致的，不作处理
        if (state !== this._state) return
        if (typeof executor !== 'function') {
            // ”乱传参数“或者”没有对应处理方案“的话，状态、数据穿透给下一个任务
            state === MyPromise._stateKeys.FULFILLED ? resolve(this._value) : reject(this._value)
        } else {
            this._addMicroStack(() => {
                try {
                    const newValue = executor(this._value)
                    if (MyPromise.isPromise(newValue)) {
                        // 返回的是一个promiseA+规范，新任务的状态和数据与传入的任务对象保持一致
                        newValue.then(resolve, reject)
                    } else {
                        resolve(newValue)
                    }
                } catch (error) {
                    reject(error)
                }
            })
        }
    }

    /**
     * 是否符合Promise A+规范，符合promiseA+规范，就认为是promise对象
     * @param {Promise} pro 
     * @returns {boolean}
     */
    static isPromise(pro) {
        // 使其永远返回的是布尔值true或者false
        return !!(pro && typeof pro === 'object' && typeof pro.then === 'function')
    }

    /**
     * 后续新的任务函数，可以继续链式调用
     * 注意：并非立即执行的。先存起来，等待当前任务触发时机，变成已决状态，推入微队列执行栈
     * then的作用效果，类似注册着，以后用
     * @param {Function} onResolved 
     * @param {Function} onRejected 
     */
    then(onResolved, onRejected) {
        return new MyPromise((resolve, reject) => {
            this._addProToStack({
                state: MyPromise._stateKeys.FULFILLED,
                executor: onResolved,
                resolve,
                reject
            })
            this._addProToStack({
                state: MyPromise._stateKeys.REJECTED,
                executor: onRejected,
                resolve,
                reject
            })
            // console.log(this._handlerStack);
            this._clearHandlerStackPushMicroStack()
        })
    }

}

# 完整版（拓展）

除了then，还有catch、finally实例方法，resolve、reject、all、allSettled、race静态方法

class MyPromise {

    /**
     * state的状态集合，不可修改
     */
    static _stateKeys = {
        PENDING: 'pending',
        FULFILLED: 'fulfilled',
        REJECTED: 'rejected'
    }

    /**
     * 任务状态
     * 规范：以下划线开头的变量，默认为私有变量，不建议外部人员使用
     */
    _state = MyPromise._stateKeys.PENDING

    /**
     * 任务状态的值
     */
    _value = void 0

    /**
     * 内部执行栈
     */
    _handlerStack = []


    /**
     * 构造函数，立即执行
     * @param {Function} executor 执行器
     */
    constructor(executor) {
        // 冻结state集合，不可修改
        Object.freeze(MyPromise._stateKeys)

        // console.log('开始状态', this._state);

        // 立即执行，处理执行器执行期间报错，走向失败状态
        try {
            executor(this._resolve.bind(this), this._reject.bind(this))
        } catch (error) {
            this._reject(error)
        }
    }

    /**
     * 改变任务状态
     * @param {string} newState 新状态
     * @param {any} value 任务传递的值
     * @returns
     */
    _changeState(newState, value) {
        if (this._state !== MyPromise._stateKeys.PENDING) return
        this._state = newState
        this._value = value
        this._clearHandlerStackPushMicroStack()
    }

    /**
     * 任务走向成功，需要多处理参数的值类型
     * @param {any} data
     */
    _resolve(data) {
        if (MyPromise.isPromise(data)) {
            data.then((val) => {
                this._resolve(val)
            }, (reason) => {
                this._reject(reason)
            })
        } else {
            this._changeState(MyPromise._stateKeys.FULFILLED, data)
        }
    }

    /**
     * 任务走向失败，不需要多处理
     * @param {any} data
     */
    _reject(reason) {
        this._changeState(MyPromise._stateKeys.REJECTED, reason)
    }

    /**
     * 添加任务对象到执行栈
     * @param { {
     *  state:string,
     *  executor:Function,
     *  resolve:Function,
     *  reject:Function
     * } } pro 任务对象
     */

        this._handlerStack.push(pro)
    }

    /**
     * 添加任务执行函数到微队列去
     * 参考vue源码的写法
     * @param {Function} callback
     */
    _addMicroStack(callback) {

        if (globalThis.process && globalThis.process.nextTick) {
            // 判断node环境
            process.nextTick(callback)
        } else if (MutationObserver) {
            // 浏览器环境有没有MutationObserver函数
            const p = document.createElement('p')
            const observer = new MutationObserver(callback)
            observer.observe(p, {
                childList: true // 观察该元素内部的变化
            })
            p.textContent = '1'
        } else {
            // 下下之策，推入的是宏队列
            setTimeout(callback, 0)
        }
    }

    /**
     * 清空内部执行栈，将执行器推入微队列
     *
     * 每次任务已决，相对的所有下一级任务执行就都可以加入微队列
     */
    _clearHandlerStackPushMicroStack() {
        if (this._state === MyPromise._stateKeys.PENDING) return
        for (const p of this._handlerStack) {
            this._pushMicroStack(p)
        }
        this._handlerStack = []
    }

    /**
     * 推入微队列
     * @param {{
     *  state:string,
     *  executor:Function,
     *  resolve:Function,
     *  reject:Function
     * }} handler
     */
    _pushMicroStack(handler) {
        const { state, executor, resolve, reject } = handler
        // 过滤状态不一致的，不作处理
        if (state !== this._state) return
        if (typeof executor !== 'function') {
            // ”乱传参数“或者”没有对应处理方案“的话，状态、数据穿透给下一个任务
            state === MyPromise._stateKeys.FULFILLED ? resolve(this._value) : reject(this._value)

            this._addMicroStack(() => {
                try {
                    const newValue = executor(this._value)
                    if (MyPromise.isPromise(newValue)) {
                        // 返回的是一个promiseA+规范，新任务的状态和数据与传入的任务对象保持一致
                        newValue.then(resolve, reject)
                    } else {
                        resolve(newValue)
                    }
                } catch (error) {
                    reject(error)
                }
            })
        }
    }

    /**
     * 是否符合Promise A+规范，符合promiseA+规范，就认为是promise对象
     * @param {Promise} pro
     * @returns {boolean}
     */
    static isPromise(pro) {
        // 使其永远返回的是布尔值true或者false
        return !!(pro && typeof pro === 'object' && typeof pro.then === 'function')
    }

    /**
     * 后续新的任务函数，可以继续链式调用
     * 注意：并非立即执行的。先存起来，等待当前任务触发时机，变成已决状态，推入微队列执行栈
     * then的作用效果，类似注册着，以后用
     * @param {Function} onResolved
     * @param {Function} onRejected
     */
    then(onResolved, onRejected) {
        return new MyPromise((resolve, reject) => {
            this._addProToStack({
                state: MyPromise._stateKeys.FULFILLED,
                executor: onResolved,
                resolve,
                reject
            })
            this._addProToStack({
                state: MyPromise._stateKeys.REJECTED,
                executor: onRejected,
                resolve,
                reject
            })
            // console.log(this._handlerStack);
            this._clearHandlerStackPushMicroStack()
        })
    }

    /**
     * 拓展1：catch
     * catch捕获错误的方法，严格来说不算是PromiseA+的必须范畴了，属于拓展部分
     * @param {Function} onRejected
     */
    catch(onRejected) {
        return this.then(null, onRejected)
    }


    /**
     * 拓展2：finally
     * 已决状态下就会执行，无论成功失败。且获得的任务对象的状态和数据在”已决执行函数，无代码执行错误“的情况下，与上一个任务对象保持一致
     * @param {Function} onSettled 已决执行函数，没有参数，返回值也没啥用
     */
    finally(onSettled) {
        return this.then((data) => {
            onSettled()
            return data
        }, (reason) => {
            onSettled()
            throw reason;
        })
    }

    /**
     * 返回一个成功状态的对象，但是参数是promise对象，则原封不动返回
     * 分三种情况处理考虑：
     * 1. 参数是自己写的具备PromiseA+规范且拓展了不少API的MyPromise对象
     * 2. 参数是其他人写的，只符合promiseA+规范的LikePromise对象，或者es6的Promise对象（稳妥点，统一处理）
     * 3. 参数是普通变量
     * @param {any} value
     */
    static resolve(value) {
        if (value instanceof MyPromise) {
            return value
        }
        return new MyPromise((resolve, reject) => {
            if (MyPromise.isPromise(value)) {
                // 参数只是符合PromiseA+规范的LikePromise对象，或者es6的Promise对象
                value.then(resolve, reject)
            } else {
                resolve(value)
            }
        })
    }


    /**
     * 返回一个失败状态的对象，其他情况不处理
     * @param {any} value
     * @returns
     */
    static reject(value) {
        return new MyPromise((resolve, reject) => {
            reject(value)
        })
    }

    /**
     * 当任务状态全为成功时，返回一个成功状态的Promise，数据为数组，且数组的每一项，对应的是参数迭代器的每一个任务对象的resolve值；
     * 若有一个失败，则返回一个失败状态的Promise，原因（数据）为失败的那个reject值就可以了
     * @param {MyPromise[]} iterable 每一项都是一个MyPromise对象，iterable是迭代器，迭代器可以是数组、伪数组、迭代器
     * @returns
     */
    static all(iterable) {
        const resultArr = []
        let loopCount = 0 // 迭代器循环的总数，即迭代器中的promise的总数
        let fulfillCount = 0 // 迭代器中的promise，已完成的个数
        return new MyPromise((resolve, reject) => {
            try {
                for (const iterator of iterable) {
                    const i = loopCount
                    loopCount++;
                    MyPromise.resolve(iterator)
                        .then((data) => {
                            resultArr[i] = data
                            fulfillCount++;
                            if (fulfillCount === loopCount) {
                                // 最后一个完成了，出口
                                resolve(resultArr)
                            }
                        }, reject)
                }
                if (loopCount === 0) {
                    resolve(resultArr)
                }
            } catch (error) {
                // 对于传参不是迭代器的情况
                reject(error)
            }
        })
    }

    /**
     * 当任务状态全为已决时，返回一个Promise对象，且其已决状态下的数据为数组，数组的每一项，对应的是参数迭代器的每一个任务对象的resolve或者reject的值
     *
     * 头脑风暴：
     * 逻辑和代码，都和静态方法all的很相似。
     * 那能不能抽离呢？（抽离出来的部分能取名字吗？想不到合适的名字，或许不适合抽离出来，也可能你没练到家）
     * 那能不能用在allSettled里面，变着法子使用all呢？
     * 实在不行，才是粗暴的cv大法修改。以上的俩种方法思路，都可以实现“高内聚，低耦合”，还能方便扩展维护！！！
     * @param {MyPromise[]} iterable 每一项都是一个MyPromise对象，iterable不算数组，是迭代器
     * @returns
     */
    static allSettled(iterable) {
        const resultArr = []
        for (const iterator of iterable) {
            resultArr.push(
                MyPromise.resolve(iterable)
                    .then(data => ({
                        status: MyPromise._stateKeys.FULFILLED,
                        value: data
                    }), reason => ({
                        status: MyPromise._stateKeys.REJECTED,
                        reason
                    }))
            )
        }
        return MyPromise.all(resultArr)
    }

    /**
     * 方法返回一个 promise，一旦迭代器中的某个 promise 解决或拒绝，返回的 promise 就会解决或拒绝。
     * @param {MyPromise[]} iterable
     * @returns
     */
    static race(iterable) {
        return new MyPromise((resolve, reject) => {
            for (const iterator of iterable) {
                MyPromise.resolve(iterator).then(resolve, reject)
            }
        })
    }

}
