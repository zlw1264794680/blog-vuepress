浅拷贝与深拷贝

zlw
2023-03-14
  • JS

# 浅拷贝

利用 Object.assign 实现浅拷贝

const obj = {
    name: 'af',
    age: 18,
    children: [
        {
            name: 'dt',
            age: 1
        }
    ]
}
const arr = [obj]
const fn = function () { }
const n = 1
function shallowClone(param) {
    if (typeof param !== "object" || param === null) {
        return param;
    }
    const ret = Array.isArray(param) ? [] : {}
    return Object.assign(ret, param)
}
console.log(shallowClone(obj)); // {name: 'af', age: 18, children: Array(1)}
console.log(shallowClone(arr)); // [{name: 'af', age: 18, children: Array(1)}]
console.log(shallowClone(fn)); // ƒ () { }
console.log(shallowClone(n)); // 1

# 深拷贝

# JSON.parse(JSON.stringify(obj))

JSON.parse(JSOn.stringify(obj))是通过JSON.stringify()把对象序列化成字符串，然后再把字符串转换成新对象。 新的对象和原对象没有关联。因为它是根据字符串转换创建。

JSON.stringify()：将对象转成 JSON 字符串。

JSON.parse()：将字符串解析成对象。

但是这个方法存在一定的局限性，具体包括：

  • 不能存放函数或者 undefined，否则会丢失函数或者 undefined；
  • 不要存放时间对象，否则会变成字符串形式；
  • 不能存放 RegExp、Error 对象，否则会变成空对象；
  • 不能存放 NaN、Infinity、-Infinity，否则会变成 null；

具体来说就是 JavaScript 和 JSON 存在差异，两者不兼容的就会出问题。

function deepClone(param) {
    if (typeof param !== "object" || param === null) {
        return param;
    }
    return JSON.parse(JSON.stringify(param))
}
const obj = {
    name: undefined,
    getName: function () { },
    birthday: new Date(),
    validate: new RegExp(),
    error: new Error(),
    nan: NaN,
    Infinity: Infinity,
    Infinity2: -Infinity
}
console.log(deepClone(obj));

# 递归实现深拷贝

const obj = {
    name: 'af',
    age: 18,
    children: [
        {
            name: 'dt',
            age: 1
        }
    ]
}
const arr = [obj]
const fn = function () { }
const n = 1
function deepClone(param) {
    if (typeof param !== "object" || param === null) {
        return param;
    }
    const ret = Array.isArray(param) ? [] : {}
    for (const key in param) {
        if (Object.hasOwnProperty.call(param, key)) {
            ret[key] = deepClone(param[key]);
        }
    }
    return ret
}
console.log(deepClone(obj)); // {name: 'af', age: 18, children: Array(1)}
console.log(deepClone(arr)); // [{name: 'af', age: 18, children: Array(1)}]
console.log(deepClone(fn)); // ƒ () { }
console.log(deepClone(n)); // 1
